# Ref

react ref

    1. 使用ref在具体的dom元素上或者class组件上
// 1. 直接创建
const ref = React.createRef();
// ref ==> { currnt: dom }
<div ref={ref}></div>

// 2. 使用回调函数（注意：直接在元素上使用回调函数，会两次渲染，一次null，第二次dom，没多大影响）
// this.ref ==> dom
<div ref={node => this.ref = node}></div>
// or ---------------------- or
constructor(props) {
  super(props);
  this.ref = null;
  this.refCb = node => this.ref = node;
}
<div ref={this.refCb}></div>
    1. forwardRef
/* 
 * 在类组件中获取ref可以使用React.forwardRef()
 * 高阶组件也是如此(具体元素转发一次，高阶组件中转发一次)
 */
React.forwareRef((props, ref) => {
  return <Component ref={ref} {...props}>;
})
